#!/bin/bash


es_archivo() {
    if [ -f "$1" ]; then
        echo "$1 es un archivo."
        return 0
    fi
    return 1
}


es_directorio() {
    if [ -d "$1" ]; then
        echo "$1 es un directorio."
        return 0
    fi
    return 1
}

if [ -z "$1" ]; then
    echo "ingrese un par√°metro."
elif [ -e "$1" ]; then
    if ! es_archivo "$1"; then
        if ! es_directorio "$1"; then
            echo "$1 existe pero no es ni archivo ni directorio."
        fi
    fi
else
    echo "$1 no existe."
fi
